-SOLUCION EN OBJETOS

class Prenda {
    var precio;
    var estado;
    
    method precioVenta(){
      return estado.modificarPrecio(precio);
    }
}

object nueva {
  
    method modificarPrecio(precio){
      return precio;
  }
}

object promocion {
    var descuento;
    
    method modificarPrecio(precio) {
      return precio - descuento;
    }
}

object liquidacion {
    
    method modificarPrecio(precio) {
      return precio * 0.5;
    }
}
